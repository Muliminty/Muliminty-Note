## 生产配置

### 完整配置

```javascript
require('esbuild').build({
  entryPoints: ['src/index.js'],
  bundle: true,
  outfile: 'dist/bundle.js',
  
  // 压缩
  minify: true,
  minifyWhitespace: true,
  minifyIdentifiers: true,
  minifySyntax: true,
  
  // Source Map
  sourcemap: true,
  
  // Tree Shaking
  treeShaking: true,
  
  // 目标环境
  target: 'es2020',
  
  // 格式
  format: 'esm',
  platform: 'browser'
})
```

## 性能优化

### 代码分割

```javascript
{
  splitting: true,
  format: 'esm',
  outdir: 'dist'
}
```

### Tree Shaking

```javascript
{
  treeShaking: true,
  bundle: true
}
```
