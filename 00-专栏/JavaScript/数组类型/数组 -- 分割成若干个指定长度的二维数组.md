/**

 * @param {*} array
 * @param {*} num
 * @return {*}

const ArrayCutting = (array, num) => {
  // 将数组按照指定的长度分割成多个子数组
  const result = Array.from({ length: Math.ceil(array.length / num) }, (_, i) => {
    // 计算子数组的起始位置
    const start = i * num;
    // 计算子数组的结束位置
    const end = start + num;
    // 过滤掉空元素
    return array.slice(start, end).filter(Boolean);
  });
  // 返回分割后的子数组
  return result;
};
